create table Users
(
    id       int GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    fullName varchar(100) UNIQUE,
    email    varchar(100)
);

select *
from Users;

create table Posts
(
    id        int GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    title     varchar(100),
    content   varchar(100),
    createdAt timestamp,
    updatedAt timestamp,
    user_id   int REFERENCES Users (id)
);

select *
from Posts;
